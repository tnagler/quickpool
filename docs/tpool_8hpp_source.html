69  {

80  }

107 class RingBuffer
108 {
109  public:
110  explicit RingBuffer(size_t capacity)
111  : buffer_{ std::unique_ptr<T[]>(new T[capacity]) }
112  , capacity_{ capacity }
113  , mask_{ capacity - 1 }
114 
115  {
116  if (capacity_ & (capacity_ - 1))
117  throw std::runtime_error("capacity must be a power of two");
118  }
119 
120  size_t capacity() const { return capacity_; }
121 
122  void set_entry(size_t i, T val) { buffer_[i & mask_] = val; }
123 
124  T get_entry(size_t i) const { return buffer_[i & mask_]; }
125 
126  RingBuffer<T>* enlarged_copy(size_t bottom, size_t top) const
127  {
128  RingBuffer<T>* new_buffer = new RingBuffer{ 2 * capacity_ };
129  for (size_t i = top; i != bottom; ++i)
130  new_buffer->set_entry(i, this->get_entry(i));
131  return new_buffer;
132  }
133 
134  private:

136  size_t capacity_;
137  size_t mask_;

152 class TaskQueue
153 {
154  using Task = std::function<void()>;
155 
156  public:
159  TaskQueue(size_t capacity = 256)
160  : buffer_{ new RingBuffer<Task*>(capacity) }
161  {}
162 
163  ~TaskQueue() noexcept
164  {
165  // must free memory allocated by push(), but not deallocated by pop()
166  auto buf_ptr = buffer_.load();
167  for (int i = top_; i < bottom_.load(m_relaxed); ++i)
168  delete buf_ptr->get_entry(i);
169  delete buf_ptr;
170  }
171 
172  TaskQueue(TaskQueue const& other) = delete;
173  TaskQueue& operator=(TaskQueue const& other) = delete;
174 
176  bool empty() const
177  {
178  return (bottom_.load(m_relaxed) <= top_.load(m_relaxed));
179  }
180 
182  void clear()
183  {
184  std::lock_guard<std::mutex> lk(mutex_); // prevents concurrent push
185  auto buf_ptr = buffer_.load();
186  auto b = bottom_.load(m_relaxed);
187  int t;
188  while (true) {
189  // try until we can set top = bottom; this might fail if someone
190  // pops concurrently
191  t = top_.load(m_relaxed);
192  if (top_.compare_exchange_weak(t, b, m_release, m_relaxed))
193  break;
194  }
195  for (int i = t; i < b; ++i)
196  delete buf_ptr->get_entry(i); // free memory for unpopped items
197  }
198 
201  bool try_push(Task&& task)
202  {
203  {
204  // must hold lock in case of multiple producers, abort if already
205  // taken, so we can check out next queue
206  std::unique_lock<std::mutex> lk(mutex_, std::try_to_lock);
207  if (!lk)
208  return false;
209  this->push_unsafe(std::forward<Task>(task));
210  }
211  cv_.notify_one();
212  return true;
213  }
214 
216  void force_push(Task&& task)
217  {
218  {
219  // must hold lock in case of multiple producers
220  std::lock_guard<std::mutex> lk(mutex_);
221  this->push_unsafe(std::forward<Task>(task));
222  }
223  cv_.notify_one();
224  }
225 
227  void push_unsafe(Task&& task)
228  {
229  auto b = bottom_.load(m_relaxed);
230  auto t = top_.load(m_acquire);
231  RingBuffer<Task*>* buf_ptr = buffer_.load(m_relaxed);
232 
233  if (static_cast<int>(buf_ptr->capacity()) < (b - t) + 1) {
234  // buffer is full, create enlarged copy before continuing
235  old_buffers_.emplace_back(
236  exchange(buf_ptr, buf_ptr->enlarged_copy(b, t)));
237  buffer_.store(buf_ptr, m_relaxed);
238  }
239 
240  buf_ptr->set_entry(b, new Task{ std::forward<Task>(task) });
241  bottom_.store(b + 1, m_release);
242  }
243 
245  bool try_pop(Task& task)
246  {
247  auto t = top_.load(m_acquire);
248  std::atomic_thread_fence(m_seq_cst);
249  auto b = bottom_.load(m_acquire);
250 
251  if (t < b) {
252  // must load task pointer before acquiring the slot, because it
253  // could be overwritten immediately after
254  auto task_ptr = buffer_.load(m_acquire)->get_entry(t);
255 
256  if (top_.compare_exchange_strong(t, t + 1, m_seq_cst, m_relaxed)) {
257  task = std::move(*task_ptr); // won race, get task
258  delete task_ptr; // fre memory allocated in push_unsafe()
259  return true;
260  }
261  }
262  return false; // queue is empty or lost race
263  }
264 
265  void wait()
266  {
267  std::unique_lock<std::mutex> lk(mutex_);
268  cv_.wait(lk, [this] { return !this->empty() || stopped_; });
269  }
270 
271  void stop()
272  {
273  {
274  std::lock_guard<std::mutex> lk(mutex_);
275  stopped_ = true;
276  }
277  cv_.notify_all();
278  }
279 
280  private:
281  alignas(64) std::atomic_int top_{ 0 };
282  alignas(64) std::atomic_int bottom_{ 0 };
283  alignas(64) std::atomic<RingBuffer<Task*>*> buffer_{ nullptr };
284  std::vector<std::unique_ptr<RingBuffer<Task*>>> old_buffers_;
285  std::mutex mutex_;
286  std::condition_variable cv_;
287  std::atomic<bool> stopped_;
288 
289  // convenience aliases
290  static constexpr std::memory_order m_relaxed = std::memory_order_relaxed;
291  static constexpr std::memory_order m_acquire = std::memory_order_acquire;
292  static constexpr std::memory_order m_release = std::memory_order_release;
293  static constexpr std::memory_order m_seq_cst = std::memory_order_seq_cst;
294 };

310  template<typename Task>
311  void push(Task&& task)
312  {
313  for (size_t count = 0; count < num_queues_ * 20; count++) {
314  if (queues_[push_idx_++ % num_queues_].try_push(task))
315  return;
316  }
317  queues_[push_idx_++ % num_queues_].force_push(task);
318  }
319 
320  template<typename Task>
321  bool try_pop(Task& task, size_t worker_id = 0)
322  {
323  for (size_t k = 0; k <= num_queues_; k++) {
324  if (queues_[(worker_id + k) % num_queues_].try_pop(task))
325  return true;
326  }
327  return false;
328  }
329 
330  void clear()
331  {
332  for (auto& q : queues_)
333  q.clear();
334  }

337 
338  void stop()
339  {
340  for (auto& q : queues_)
341  q.stop();
342  stopped_ = true;
343  }

351 class ThreadPool
352 {
353  public:
355  ThreadPool()
356  : ThreadPool(std::thread::hardware_concurrency())
357  {}
358 
362  explicit ThreadPool(size_t n_workers)
363  : task_manager_{ n_workers }
364  {
365  for (size_t id = 0; id < n_workers; ++id) {
366  workers_.emplace_back([this, id] {
367  std::function<void()> task;
368  while (!task_manager_.stopped()) {
369  task_manager_.wait_for_jobs(id);
370  do {
371  // inner while to save a few cash misses calling empty()
372  while (task_manager_.try_pop(task, id))
373  this->execute_safely(task);
374  } while (!todo_list_.empty());
375  }
376  });
377  }
378  }
379 
380  ~ThreadPool()
381  {
382  task_manager_.stop();
383  for (auto& worker : workers_) {
384  if (worker.joinable())
385  worker.join();
386  }
387  }

404  template<class Function, class... Args>
405  void push(Function&& f, Args&&... args)
406  {
407  if (workers_.size() == 0)
408  return f(args...);
409  todo_list_.add();
410  task_manager_.push(
411  std::bind(std::forward<Function>(f), std::forward<Args>(args)...));
412  }
413 
419  template<class Function, class... Args>
420  auto async(Function&& f, Args&&... args)
421  -> std::future<decltype(f(args...))>
422  {
423  auto pack =
424  std::bind(std::forward<Function>(f), std::forward<Args>(args)...);
425  using pack_t = std::packaged_task<decltype(f(args...))()>;
426  auto task_ptr = std::make_shared<pack_t>(std::move(pack));
427  this->push([task_ptr] { (*task_ptr)(); });
428  return task_ptr->get_future();
429  }
430 
432  void wait() { todo_list_.wait(); }
433 
434  private:
435  void execute_safely(std::function<void()>& task)
436  {
437  try {
438  task();
439  todo_list_.cross();
440  } catch (...) {
441  todo_list_.clear(std::current_exception());
442  }
443  }
444 
445  detail::TaskManager task_manager_;
446  TodoList todo_list_{ 0 };
447  std::vector<std::thread> workers_;
448 };
